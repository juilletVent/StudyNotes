## 索引访问类型 Indexed Access Types

可以使用索引访问类型来获取对象中的特定属性：

```ts
type Person = { age: number; name: string; alive: boolean };
// 单独拿到属性类型
type Age = Person["age"]; // number

// 可以搭配联合体使用
type I1 = Person["age" | "name"];
// I1 = string | number;

// 可以借助keyof生成全部索引
type I2 = Person[keyof Person];
// I2 = string | number | boolean;

type AliveOrName = "alive" | "name";
// I3 = Person[AliveOrName];

// 另外一些常用的用法：
const MyArray = [
  { name: "Alice", age: 15 },
  { name: "Bob", age: 23 },
  { name: "Eve", age: 38 },
];

type Person = (typeof MyArray)[number];
type Age = (typeof MyArray)[number]["age"];
// or
type Age2 = Person["age"];
```

### 搭配字符串模式进行前缀Key类型过滤

```ts
type Person = { age: number; name: string; alive: boolean };

// 搭配字符串模板过滤特定前缀的key
type PrefixKeys<T> = {
  [K in keyof T]: K extends `a${string}` ? K : never;
}[keyof T];
type PrefixA = PrefixKeys<Person>; // age | alive

const obj: PrefixA = "age"; // ok
```

**重点：类型是类型，值是值，类型相关的验证与操作位于编译阶段，而值则位于运行阶段，不能混用**

```ts
const key = "age";
// ERROR：Type 'key' cannot be used as an index type.
// 'key' refers to a value, but is being used as a type here. Did you mean 'typeof key'?
// 错误提示：key是值，不能作为索引类型
type Age = Person[key];

// 我们可以借助类型别名完成重构：
type Key = "age"; // 我不是值，我是一个类型 AHA~
type Age = Person[Key];
```
